# torch is NOT listed here because the macOS x86_64 CPU wheel must come from the
# PyTorch CPU index, not PyPI. Install it first, then this file:

#   pip install -r requirements-mac.txt

# --- Core deep-learning stack (CPU build is sufficient: the backbone is frozen) ---
torch>=2.2

# --- Wildlife dataset access (download + standardized metadata dataframe) ---
# NOTE: we intentionally do NOT depend on `wildlife-tools` — it hard-requires

# "Numpy is not available" under NumPy 2. Lift this cap only if torch is upgraded.
numpy>=1.26,<2
pandas>=2.0
pydantic>=2.6
pydantic-settings>=2.2

# --- Metrics + visualisation ---
